CREATE PROCEDURE [dbo].[prc_Empleado_dl]
(
	@EmpleadoId INT,
	@ErrorId INT OUTPUT,
	@Error VARCHAR(MAX) OUTPUT
)AS
BEGIN
SET NOCOUNT ON
BEGIN TRY

	IF(NOT EXISTS(SELECT TOP 1 1 FROM tbl_Empleado WHERE EmpleadoId = @EmpleadoId))
	BEGIN
		SET @ErrorId = -1
		SET @Error = 'El EmpleadoId no existe'
		RETURN
	END

	UPDATE tbl_Empleado 
	SET			FechaBaja = GETDATE()
	WHERE	EmpleadoId = @EmpleadoId

	SET @ErrorId = 0
	SET @Error = ''

	SELECT * FROM tbl_Empleado
	WHERE	EmpleadoId = @EmpleadoId

END TRY
BEGIN CATCH
	SET @ErrorId = ERROR_NUMBER()
	SET @Error = 'Ocurrio un error inerperado. '+ERROR_MESSAGE()
END CATCH
END

CREATE PROCEDURE [dbo].[prc_Empleado_in]
(
	@Nombre VARCHAR(50),
	@ApellidoPaterno VARCHAR(50),
	@ApellidoMaterno VARCHAR(50),
	@FechaNacimiento DATETIME,
	@Genero VARCHAR(10),
	@EstadoCivil VARCHAR(30),
	@RFC VARCHAR(13),
	@Direccion VARCHAR(MAX),
	@Email VARCHAR(100),
	@Telefono BIGINT,
	@Puesto VARCHAR(100),
	@ErrorId INT OUTPUT,
	@Error VARCHAR(MAX) OUTPUT
)AS
BEGIN
SET NOCOUNT ON
BEGIN TRY
	
	INSERT INTO tbl_Empleado (
							Nombre,
							ApellidoPaterno,
							ApellidoMaterno,
							Edad,
							FechaNacimiento,
							Genero,
							EstadoCivil,
							RFC,
							Direccion,
							Email,
							Telefono,
							Puesto,
							FechaAlta)
	SELECT		@Nombre,
				@ApellidoPaterno,
				@ApellidoMaterno,
				Edad = DATEDIFF(MM,@FechaNacimiento,GETDATE())/12,
				@FechaNacimiento,
				@Genero,
				@EstadoCivil,
				@RFC,
				@Direccion,
				@Email,
				@Telefono,
				@Puesto,
				FechaAlta = GETDATE()

	SET @ErrorId = 0
	SET @Error = ''

	SELECT		EmpleadoId,
				Nombre,
				ApellidoPaterno,
				ApellidoMaterno,
				Edad,
				FechaNacimiento,
				Genero,
				EstadoCivil,
				RFC,
				Direccion,
				Email,
				Telefono,
				Puesto,
				FechaAlta,
				FechaBaja
		FROM	tbl_Empleado	
		WHERE	EmpleadoId = @@Identity

END TRY
BEGIN CATCH
	SET @ErrorId = ERROR_NUMBER()
	SET @Error = 'Ocurrio un error inerperado. '+ERROR_MESSAGE()
END CATCH
END

CREATE PROCEDURE [dbo].[prc_Empleado_up]

	@Email VARCHAR(100),
	@Telefono BIGINT,
	@Puesto VARCHAR(100),
	@FechaBaja DATETIME = NULL,
	@EstadoCivil VARCHAR(30),
	@Direccion VARCHAR(MAX),

BEGIN

		SET @ErrorId = -1

	UPDATE tbl_Empleado

				EstadoCivil = @EstadoCivil,

	SET @Error = ''

	SELECT * FROM tbl_Empleado
	WHERE	EmpleadoId = @EmpleadoId

END TRY
BEGIN CATCH
	SET @ErrorId = ERROR_NUMBER()
	SET @Error = 'Ocurrio un error inerperado. '+ERROR_MESSAGE()
END CATCH
END
